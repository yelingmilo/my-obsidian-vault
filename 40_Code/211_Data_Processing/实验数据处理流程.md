## 将实验视频解压识别
主要是会生成 3 个文本文件和照片：
时间文件 DefaultVideo_vnum. Txt 文件
粒子位置文件 ps. Txt 文件总共十列，每列为[y x	mass	size	ecc	signal	raw_mass	ep	frame	particle] 
光阱位置文件 trap_vnum.Txt 文件前三列分别为 x, y, strength
这些数据保存在 data 文件夹中。
下面是它的代码：
```python
import os
import struct
from PIL import Image
import numpy as np
from concurrent.futures import ProcessPoolExecutor
from functools import partial
import matplotlib as mpl
import matplotlib.pyplot as plt
import pandas as pd
from pandas import DataFrame, Series
import pims
import trackpy as tp


@pims.pipeline
def gray(image):
    return image[:, :]  # Take just the green channel


def read_twv_header(file):
    header_format = '<HHHII'
    header_size = struct.calcsize(header_format)
    header_data = file.read(header_size)
    return struct.unpack(header_format, header_data)


def read_frame_properties(file):
    frame_properties_format = '<HHHHHHHdddH'
    frame_properties_size = struct.calcsize(frame_properties_format)
    frame_properties_data = file.read(frame_properties_size)
    return struct.unpack(frame_properties_format, frame_properties_data)


def read_frame_header(file):
    frame_header_format = '<I14f'
    frame_header_size = struct.calcsize(frame_header_format)
    frame_header_data = file.read(frame_header_size)
    return struct.unpack(frame_header_format, frame_header_data)


def read_frame_bitmap(file, width, height, bytes_per_pixel):
    bitmap_size = width * height * bytes_per_pixel
    bitmap_data = file.read(bitmap_size)
    actual_size = len(bitmap_data)
    print(f'Expected bitmap size: {bitmap_size}, Actual bitmap size: {actual_size}')
    if actual_size!= bitmap_size:
        raise ValueError(f'Bitmap size mismatch: expected {bitmap_size}, got {actual_size}')
    return np.frombuffer(bitmap_data, dtype=np.uint8).reshape((height, width, bytes_per_pixel))


def save_frame_as_image(frame_bitmap, frame_number, mode, outpath):
    print(f'Frame {frame_number} shape: {frame_bitmap.shape}, dtype: {frame_bitmap.dtype}')
    image = Image.fromarray(frame_bitmap, mode)
    image.save(os.path.join(outpath, f'frame_{frame_number:04d}.png'))


def convert_twv_to_images(twv_file_path, datapath):
    base_name = os.path.splitext(os.path.basename(twv_file_path))[0]
    outpath = os.path.join(datapath, 'v' + base_name.split('_')[-1])
    os.makedirs(outpath, exist_ok=True)

    with open(os.path.join(datapath, twv_file_path), 'rb') as file:
        twv_header = read_twv_header(file)
        num_frames = twv_header[4]
        frame_properties = read_frame_properties(file)
        bytes_per_pixel = frame_properties[6]
        width = frame_properties[4]
        height = frame_properties[5]

        with open(os.path.join(outpath, f'trap_{base_name.split("_")[-1]}.txt'), 'w') as trap_file, \
                open(os.path.join(outpath, f'{base_name}.txt'), 'w') as default_video_file:
            for frame_number in range(num_frames):
                frame_header = read_frame_header(file)
                bitmap_size = width * height * bytes_per_pixel
                bitmap_data = file.read(bitmap_size)
                frame_bitmap = np.frombuffer(bitmap_data, dtype=np.uint8).reshape((height, width))
                image = Image.fromarray(frame_bitmap, mode='L')
                image.save(os.path.join(outpath, f'frame_{frame_number:04d}.png'))
                # Write frame_header[4:9] to trap_number.txt
                trap_file.write(' '.join(map(str, frame_header[4:16])) + '\n')
                default_video_file.write(str(frame_header[2]) + '\n')


def convert_single_twv_to_images(index, datapath):
    twv_filename = f'DefaultVideo_{index}.twv'   #####视频名字设置
    try:
        convert_twv_to_images(twv_filename, datapath)
    except Exception as e:
        print(f"Error processing file {twv_filename}: {e}")


def process_images_in_directory(version, base_filepath, micron_per_pixel, feature_diameter):
    filepath = f"{base_filepath}v{version}/"
    frames = gray(pims.open(filepath + "*.png"))

    radius = int(np.round(feature_diameter / 2.0 / micron_per_pixel))
    if radius % 2 == 0:
        radius += 1
    print(f'Using a radius of {radius} px for version {version}')

    features = tp.batch(frames[:1000], radius + 2, invert=False, minmass=3000, engine='numba')
    t = tp.link(features, 11)

    # 将t转换为DataFrame类型（如果它还不是的话），以便于保存为文本文件
    if not isinstance(t, pd.DataFrame):
        t = pd.DataFrame(t)

    # 构建保存文件的完整路径
    save_path = filepath + "ps.txt"

    # 使用pandas的to_csv方法保存数据，设置分隔符为制表符（'\t'），索引设置为False以避免保存行索引
    t.to_csv(save_path, sep='\t', index=False)

    print(f"数据已成功保存到 {save_path} for version {version}")


def main():
    datapath = '/data/milo/data/251211/'
    start, end = 41,79 
    step = 1

    # 先将.twv文件转换为图像及相关文本文件
    partial_func = partial(convert_single_twv_to_images, datapath=datapath)
    with ProcessPoolExecutor(max_workers=32) as executor:
        results = list(executor.map(partial_func, range(start, end, step)))

    # 定义处理图像时的参数
    micron_per_pixel = 0.055
    feature_diameter = 2.0

    # 再对生成的图像进行处理并保存粒子追踪结果
    versions = range(start, end, step)
    for version in versions:
        process_images_in_directory(version, datapath, micron_per_pixel, feature_diameter)


if __name__ == '__main__':
    main()
```

## Link 粒子轨迹
从上面的文件中得到粒子的轨迹的信息并放到目标文件夹 Z:\milo\result\predata\中，粒子信息为 ptr 1_vnum. Txt, 每列分别为：[y x	mass	size	ecc	signal	raw_mass	ep	frame	particle] 
光阱位置信息文件 traptr 1_vnum.Txt, 每列为[x, y, frame, id]
下面是他的程序：
```Python
# -*- coding: utf-8 -*-
"""
统一的粒子分析工作流脚本 (并行优化版 v2 - 内存安全)

此脚本通过任务解耦实现高效并行处理:
1. 任务 A (追踪) 和 B (陷阱) 作为独立任务并行提交。
2. 当任务 A 成功完成后，其依赖的任务 C (ETL) 才被动态提交到池中。

[内存优化 v2]
- 任务 C (ETL) 使用 pandas.read_csv 的 'chunksize' 参数，
  避免将大型 ps.txt 文件一次性读入内存，防止 OOM。
- 增加了关于 MAX_WORKERS 的内存警告。

[v3 修正]
- 修正了 3.2 节中的并行逻辑错误。
- 使用 while 循环配合 set (active_futures) 来正确管理
  动态提交的 "任务 C"，确保 as_completed() 可以监控到它们。
"""

import os
import sys
import shutil
import traceback
import pandas as pd  # 引入 pandas
from concurrent.futures import ProcessPoolExecutor, as_completed
from particle_analyzer import ParticleAnalyzer  # 确保 particle_analyzer.py 在您的 PYTHONPATH 中

# --- 1. 核心配置 (用户应在此处修改) ---

# 实验标识
MY_DATE = 251211
VNUM_START = 1
VNUM_END = 40

# [!! 内存警告 !!]
# MAX_WORKERS 决定了有多少任务 *同时* 运行。
# 如果 run_particle_tracking (任务A) 或 run_file_etl (任务C) 非常消耗内存，
# (例如：每个任务 4GB RAM)，而您的机器总共只有 32GB RAM，
# 那么设置 MAX_WORKERS = 16 (16 * 4GB = 64GB) 将导致内存溢出。
#
# !! 请根据您机器的 (总RAM / 单个任务峰值RAM) 来调低此值 !!
MAX_WORKERS = 16 

# 基础路径
BASE_DATA_DIR = '/data/milo/data'
BASE_RESULT_DIR = '/data/milo/result/predata'

# --- 任务 A: 粒子追踪参数 (来自脚本 1) ---
TRACKING_ROI = [100, 100, 300, 300]
PARTICLE_DIAM_PX = 11
TRACKING_MINMASS = 300
TRACKING_LINK_RANGE = 40
TRACKING_MAX_FRAMES = 100000

# --- 任务 B: 陷阱分析参数 (来自脚本 2) ---
TRAP_NUM = 12
IMAGE_HEIGHT = 200

# --- 2. 任务函数定义 (已优化) ---

def run_particle_tracking(datapath, version, roi, diameter, minmass, link_range, max_frames):
    """任务 A: 运行粒子追踪"""
    # (此函数未变)
    output_file_path = os.path.join(datapath, f'v{version}', 'ps.txt')
    try:
        ParticleAnalyzer.track_particles_in_roi(
            datapath=datapath,
            version=version,
            roi_rect=roi,
            diameter_pixels=diameter,
            minmass=minmass,
            link_search_range=link_range,
            max_frames=max_frames,
            output_filename=output_file_path
        )
        return True, version, None
    except Exception as e:
        return False, version, traceback.format_exc()

def run_trap_preprocessing(date, vnum, trapnum, img_height, data_base, result_base):
    """任务 B: 运行陷阱轨迹预处理"""
    # (此函数未变)
    try:
        output_path = ParticleAnalyzer.preprocess_trap_trajectories(
            date=date,
            vnum=vnum,
            trapnum=trapnum,
            image_height=img_height,
            data_base_path=data_base,
            result_base_path=result_base
        )
        if output_path:
            return True, vnum, None
        else:
            return False, vnum, "未生成输出路径"
    except Exception as e:
        return False, vnum, traceback.format_exc()

def run_file_etl(date, vnum, data_base, result_base):
    """任务 C: 复制和转换文件 (内存优化版)"""
    try:
        src_dir = os.path.join(data_base, str(date), f"v{vnum}")
        dest_dir = os.path.join(result_base, str(date))
        os.makedirs(dest_dir, exist_ok=True)

        # --- [内存优化] ---
        # 使用 Pandas 分块处理 ps.txt -> ptr1_v...txt
        
        src_ps_file = os.path.join(src_dir, 'ps.txt')
        dest_ptr_file = os.path.join(dest_dir, f'ptr1_v{vnum}.txt')
        
        # 定义一个合理的块大小，例如一次处理100,000行
        # 这意味着内存峰值约等于10万行数据，而不是整个文件
        chunk_size = 100000 
        
        # 创建一个 TextFileReader 迭代器
        reader = pd.read_csv(
            src_ps_file, 
            delim_whitespace=True, 
            skiprows=1, 
            comment='#', 
            header=None,
            chunksize=chunk_size  # [!! 关键 !!] 指定块大小
        )
        
        # 迭代处理每个块
        is_first_chunk = True
        for chunk_df in reader:
            if is_first_chunk:
                # 第一次写入，使用 'w' 模式 (覆盖)
                chunk_df.to_csv(dest_ptr_file, index=False, header=False, mode='w')
                is_first_chunk = False
            else:
                # 后续写入，使用 'a' 模式 (追加)
                chunk_df.to_csv(dest_ptr_file, index=False, header=False, mode='a')
        
        # --- [内存优化结束] ---

        # 复制 DefaultVideo.txt -> t_v...txt (此操作内存占用很小)
        src_default_video = os.path.join(src_dir, f'DefaultVideo_{vnum}.txt')
        dest_t_v_file = os.path.join(dest_dir, f't_v{vnum}.txt')
        shutil.copy(src_default_video, dest_t_v_file)
        
        return True, vnum, None
        
    except Exception as e:
        return False, vnum, traceback.format_exc()


# --- 3. 主执行流程 (并行优化) ---

if __name__ == '__main__':
    print("=" * 60)
    print(f"启动并行分析工作流 (v2.1 - 动态任务修正版)")
    print(f"日期: {MY_DATE}")
    print(f"vnum范围: {VNUM_START} 到 {VNUM_END-1}")
    print(f"并行工作进程数: {MAX_WORKERS} (!!请注意内存限制!!)")
    print("=" * 60)
    
    vnums = list(range(VNUM_START, VNUM_END))
    
    # [优化] 结果追踪器，用于收集所有状态
    results_tracker = {
        v: {'tracking': None, 'trap': None, 'etl': None, 'errors': {}} 
        for v in vnums
    }
    
    # [优化] futures_map 用于将 future 对象映射回其任务信息
    futures_map = {}
    
    with ProcessPoolExecutor(max_workers=MAX_WORKERS) as executor:
        
        # --- 3.1 提交所有初始任务 (Task A 和 Task B) ---
        print(f"正在提交 {len(vnums) * 2} 个初始任务 (Tracking + Trap)...")
        for vnum in vnums:
            date_specific_datapath = os.path.join(BASE_DATA_DIR, str(MY_DATE))
            
            # 提交任务 A (Tracking)
            future_a = executor.submit(
                run_particle_tracking,
                datapath=date_specific_datapath,
                version=vnum,
                roi=TRACKING_ROI,
                diameter=PARTICLE_DIAM_PX,
                minmass=TRACKING_MINMASS,
                link_range=TRACKING_LINK_RANGE,
                max_frames=TRACKING_MAX_FRAMES
            )
            futures_map[future_a] = ('tracking', vnum)
            
            # 提交任务 B (Trap)
            future_b = executor.submit(
                run_trap_preprocessing,
                date=MY_DATE,
                vnum=vnum,
                trapnum=TRAP_NUM,
                img_height=IMAGE_HEIGHT,
                data_base=BASE_DATA_DIR,
                result_base=BASE_RESULT_DIR
            )
            futures_map[future_b] = ('trap', vnum)

        # ##################################################################
        # --- 3.2 处理完成的任务 (动态提交 Task C) - [已修正] ---
        # ##################################################################
        
        # [修正] 使用一个“动态平板” (set) 来跟踪所有*活动*的任务
        # 初始状态下，它包含所有A和B任务
        active_futures = set(futures_map.keys())
        total_tasks_submitted = len(active_futures)
        completed_tasks = 0
        
        print(f"开始处理 {total_tasks_submitted} 个初始任务...")

        # [修正] 使用 while 循环，只要“动态平板”不为空就继续
        while active_futures:
            
            # [修正] 等待“动态平板”上的*任何*一个任务完成
            try:
                # as_completed 会等待，直到集合中至少有一个任务完成
                future = next(as_completed(active_futures))
            except StopIteration:
                # 理论上，如果 while 循环正确，这里不应该发生
                break 

            # 从映射中获取这个任务的信息
            task_type, vnum = futures_map[future]
            completed_tasks += 1
            
            # [修正] 任务完成后，立刻将其从“动态平板”上移除
            active_futures.remove(future)

            try:
                success, _, error_msg = future.result()
            except Exception as e:
                success = False
                error_msg = f"工作流执行器异常: {e}\n{traceback.format_exc()}"

            # 更新结果追踪器
            results_tracker[vnum][task_type] = success
            if not success:
                results_tracker[vnum]['errors'][task_type] = error_msg
                print(f"[v{vnum}] ❌ 任务 {task_type} 失败. (已处理 {completed_tasks}/{total_tasks_submitted})")
            else:
                print(f"[v{vnum}] ✅ 任务 {task_type} 成功. (已处理 {completed_tasks}/{total_tasks_submitted})")

            # [优化] 关键点: 如果任务A成功，动态提交任务C
            if task_type == 'tracking' and success:
                print(f"  └─ [v{vnum}] 追踪成功，提交文件转换 (ETL) 任务...")
                future_c = executor.submit(
                    run_file_etl,
                    date=MY_DATE,
                    vnum=vnum,
                    data_base=BASE_DATA_DIR,
                    result_base=BASE_RESULT_DIR
                )
                # 把它也注册到总花名册
                futures_map[future_c] = ('etl', vnum)
                # [修正] 并把它添加到“动态平板”上，让 while 循环知道要等它
                active_futures.add(future_c)
                total_tasks_submitted += 1 # 动态更新总任务数
            
            # 如果任务A失败，标记ETL为 "skipped" (非失败，非成功)
            elif task_type == 'tracking' and not success:
                results_tracker[vnum]['etl'] = 'skipped'
                print(f"  └─ [v{vnum}] 追踪失败，跳过文件转换 (ETL).")
            
    # --- 4. 最终统计报告 ---
    print("\n" + "=" * 60)
    print("全部任务执行完毕 - 统计报告:")
    print("=" * 60)
    
    successful_vnums = []
    failed_vnums = []
    
    for vnum, data in results_tracker.items():
        # 严格定义成功：A, B, C都必须是True
        # (注意：如果A失败，C是'skipped'，所以 is_strict_success 会是 False)
        is_strict_success = data['tracking'] and data['trap'] and data['etl']
        
        if is_strict_success:
            successful_vnums.append(vnum)
        else:
            failed_vnums.append((vnum, data['errors'], data))
    
    print(f"总 vnum 数: {len(vnums)}")
    print(f"严格成功 (A, B, C均成功): {len(successful_vnums)} 个")
    print(f"失败或部分成功: {len(failed_vnums)} 个")
    
    if successful_vnums:
        print(f"\n成功的 vnums: {sorted(successful_vnums)}")
    
    if failed_vnums:
        print(f"\n失败/部分成功的 vnums:")
        for vnum, errors, data in sorted(failed_vnums):
            # 构造一个简短的状态报告
            t_status = '✅' if data['tracking'] else ('❌' if data['tracking'] is False else 'N/A')
            p_status = '✅' if data['trap'] else ('❌' if data['trap'] is False else 'N/A')
            e_status = '✅' if data['etl'] else ('⏭️' if data['etl'] == 'skipped' else ('❌' if data['etl'] is False else 'N/A'))
            
            print(f"  - v{vnum} (状态: 追踪{t_status}, 陷阱{p_status}, 转换{e_status}):")
            if not errors and e_status == '⏭️':
                print("      └─ (任务A失败导致任务C被跳过)")
            elif not errors:
                 print("      └─ (无详细错误信息)")
            for task, error_detail in errors.items():
                # 只打印错误的摘要，避免刷屏
                try:
                    error_summary = str(error_detail).strip().split('\n')[-1] # 通常 traceback 的最后一行是错误类型
                except:
                    error_summary = str(error_detail)
                print(f"      └─ {task} 失败: {error_summary}...")
    
    print("=" * 60)
```

```python
from particle_analyzer import ParticleAnalyzer
import numpy as np

matplotlib.use('Agg')

# =====================================================================
# 函数定义 (保持不变)

def gaussian(y, mean, std):
    return norm.pdf(y, mean, std)

def constrained_gmm(y_data, w1, w2, std1, std2, std3, x):
    w3 = 1.0 - w1 - w2
    if w3 < 0:
        return np.inf
    mean1 = x
    mean2 = 0.0
    mean3 = -2.0 * x
    comp1 = w1 * gaussian(y_data, mean1, std1)
    comp2 = w2 * gaussian(y_data, mean2, std2)
    comp3 = w3 * gaussian(y_data, mean3, std3)
    return comp1 + comp2 + comp3

# 循环处理
for filenum1 in tqdm(range(11, 231), desc="正在处理文件"):

    # [!!! 核心修改 !!!]

        # 检查数据是否加载成功

        first_idx = 0

        # 2. 计算斜率
        sampling_rate = 60

        slopes_array = np.array(slopes)
        slope_times = np.array(slope_times)

        if len(slopes_array) == 0:
            raise ValueError("数据不足，无法计算斜率")

        data_to_fit = -slopes_array
        
        print(f"原始数据点数: {len(data_to_fit)}")
        Q1 = np.percentile(data_to_fit, 25)
        Q3 = np.percentile(data_to_fit, 75)
        IQR = Q3 - Q1
        lower_bound = Q1 - 1.5 * IQR
        upper_bound = Q3 + 1.5 * IQR

        )

        # --- 图 (a) ---
        ax1.plot(t_shifted, -theta_shifted, label='Original $\\theta$', **STYLE_CONFIG["line_styles"]["theta_raw"])

                   fontsize=STYLE_CONFIG['font_sizes']['base'],

        ax1.set_title(f'Slope of $\\theta$ (Window: {window_seconds}s)', fontsize=STYLE_CONFIG['font_sizes']['title'])

        hist_y, bin_edges, _ = ax_hist.hist(data_to_fit_filtered, bins=50, density=True, alpha=0.6, 
                                            color=STYLE_CONFIG["line_styles"]["theta_slope"]['color'], label='data')
        bin_centers = (bin_edges[:-1] + bin_edges[1:]) / 2.0
        
        # 5. 拟合
        try:
            p0 = [0.2, 0.6, 0.1, 0.1, 0.1, 0.2]
            bounds = ([0, 0, 0.01, 0.01, 0.01, 0.01], [1, 1, 1, 1, 1, 1])

            x_plot = np.linspace(bin_centers.min(), bin_centers.max(), 1000)

        except Exception as e:
            print(f"⚠️ [警告] 约束拟合失败 (仅跳过拟合步骤): {e}")
            # 注意：这里我们只捕获拟合的错误，不让拟合失败阻止图片保存

        ax_hist.set_xlabel('Slope (rad/s)', fontsize=STYLE_CONFIG['font_sizes']['label'])
        ax_hist.set_title('Slope Distribution', fontsize=STYLE_CONFIG['font_sizes']['title'])
        
        plt.tight_layout()
        
        save_filename = f"fitW_{filenum1}.png"
        full_save_path = os.path.join(SAVE_PATH, save_filename)
        plt.savefig(full_save_path, format='png', bbox_inches='tight', dpi=STYLE_CONFIG['figure']['dpi'])

        print(f"Stats - Mean: {np.mean(slopes_array):.4f}, Filtered Mean: {np.mean(data_to_fit_filtered):.4f}")

        # 关闭当前图像
        plt.close(fig)

    # [!!! 核心修改 !!!] 
    # 捕获所有错误，打印后继续下一个循环
    except Exception as e:

        continue
```
